**`app/api/staff/people/merge/route.ts`** (POST):

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { z } from 'zod';
import { prisma } from '@/lib/prisma';
import { requireSession, ok, apiError } from '@/lib/api-response';

const MergePersonSchema = z.object({
  source_id: z.string().cuid(),
  target_id: z.string().cuid(),
});

export async function POST(request: NextRequest) {
  const sessionOrError = await requireSession('staff');
  if (sessionOrError instanceof NextResponse) return sessionOrError;
  const session = sessionOrError;

  const body = await request.json();
  const parsed = MergePersonSchema.safeParse(body);
  if (!parsed.success) return apiError('VALIDATION_ERROR', 'Invalid input', 422, parsed.error.flatten().fieldErrors);
  const { source_id, target_id } = parsed.data;

  // Cannot merge with self
  if (source_id === target_id) return apiError('MERGE_SAME', 'Source and target must be different people', 422);

  // Load both persons
  const [source, target] = await Promise.all([
    prisma.person.findFirst({ where: { id: source_id, deleted_at: null } }),
    prisma.person.findFirst({ where: { id: target_id, deleted_at: null } }),
  ]);

  if (!source) return apiError('NOT_FOUND', 'Source person not found', 404);
  if (!target) return apiError('NOT_FOUND', 'Target person not found', 404);
  if (source.anonymized_at) return apiError('PERSON_ANONYMIZED', 'Cannot merge an anonymized record', 422);
  if (target.anonymized_at) return apiError('PERSON_ANONYMIZED', 'Cannot merge into an anonymized record', 422);

  // Get all TicketPerson rows for source (need to re-point to target)
  const sourceTicketPersons = await prisma.ticketPerson.findMany({
    where: { person_id: source_id },
    select: { id: true, ticket_id: true, person_id: true, role: true },
  });

  // Build upsert operations: for each sourceTicketPerson, try to re-point to target
  // If (ticket_id, target_id) already exists → skip (avoid duplicate join row)
  const existingTargetLinks = await prisma.ticketPerson.findMany({
    where: { person_id: target_id },
    select: { ticket_id: true },
  });
  const existingTargetTicketIds = new Set(existingTargetLinks.map(l => l.ticket_id));

  // TicketPersons to migrate (no existing link to target for this ticket)
  const toMigrate = sourceTicketPersons.filter(tp => !existingTargetTicketIds.has(tp.ticket_id));
  // TicketPersons to delete (already linked to target — delete source's duplicate)
  const toDelete = sourceTicketPersons.filter(tp => existingTargetTicketIds.has(tp.ticket_id));

  // Copy non-null fields from source to target (only where target field is null)
  const fieldUpdates: Record<string, unknown> = {};
  if (!target.name && source.name) fieldUpdates.name = source.name;
  if (!target.email && source.email) fieldUpdates.email = source.email;
  if (!target.phone && source.phone) fieldUpdates.phone = source.phone;
  if (!target.notes && source.notes) fieldUpdates.notes = source.notes;

  // Build transaction operations
  const ops = [
    // Re-point migrating TicketPerson rows to target
    ...toMigrate.map(tp =>
      prisma.ticketPerson.update({
        where: { id: tp.id },
        data: { person_id: target_id },
      })
    ),
    // Delete duplicate TicketPerson rows (source had link to ticket already linked to target)
    ...toDelete.map(tp =>
      prisma.ticketPerson.delete({ where: { id: tp.id } })
    ),
    // Update target with copied fields (if any)
    ...(Object.keys(fieldUpdates).length > 0
      ? [prisma.person.update({ where: { id: target_id }, data: fieldUpdates })]
      : []),
    // Soft-delete source
    prisma.person.update({
      where: { id: source_id },
      data: {
        deleted_at: new Date(),
        merged_into_id: target_id,
      },
    }),
    // TicketHistory notes on affected tickets
    ...toMigrate.map(tp =>
      prisma.ticketHistory.create({
        data: {
          ticket_id: tp.ticket_id,
          actor_id: session.user.id,
          action: 'PERSON_LINKED',
          note: `Person records merged by ${session.user.username}`,
        },
      })
    ),
  ];

  await prisma.$transaction(ops);

  return ok({ target_person_id: target_id, tickets_relinked: toMigrate.length });
}
```

**`app/api/staff/people/[id]/anonymize/route.ts`** (PATCH):

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { prisma } from '@/lib/prisma';
import { requireSession, ok, apiError } from '@/lib/api-response';

export async function PATCH(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  const sessionOrError = await requireSession('staff');
  if (sessionOrError instanceof NextResponse) return sessionOrError;
  const session = sessionOrError;

  const { id } = await params;

  // Load person
  const person = await prisma.person.findFirst({ where: { id, deleted_at: null } });
  if (!person) return apiError('NOT_FOUND', 'Person not found', 404);

  // Cannot re-anonymize
  if (person.anonymized_at) return apiError('ALREADY_ANONYMIZED', 'This record has already been anonymized', 409);

  // Transaction: null all PII fields + set anonymized_at
  // Per FRD F05: null name, email, phone, notes; set anonymized_at = now()
  await prisma.$transaction([
    prisma.person.update({
      where: { id },
      data: {
        name: null,
        email: null,
        phone: null,
        notes: null,
        anonymized_at: new Date(),
      },
    }),
    // TicketHistory audit entries on all linked tickets
    // Use raw create for each — find linked ticket IDs first
    // Actually Prisma transactions don't support dynamic lookups before operations.
    // Solution: do lookup BEFORE transaction, then include creates in transaction.
  ]);

  // The above transaction doesn't include TicketHistory. Fix: find linked tickets first, then run full transaction.
  // Replace the above with:
  const ticketLinks = await prisma.ticketPerson.findMany({
    where: { person_id: id },
    select: { ticket_id: true },
  });

  await prisma.$transaction([
    prisma.person.update({
      where: { id },
      data: {
        name: null,
        email: null,
        phone: null,
        notes: null,
        anonymized_at: new Date(),
      },
    }),
    ...ticketLinks.map(tp =>
      prisma.ticketHistory.create({
        data: {
          ticket_id: tp.ticket_id,
          actor_id: session.user.id,
          action: 'PERSON_ANONYMIZED' as const,
          note: `Person record anonymized by ${session.user.username}`,
        },
      })
    ),
  ]);

  return new NextResponse(null, { status: 204 });
}
```

**NOTE on TicketHistory action values**: The Prisma schema's `TicketHistory.action` is a `String` type (not an enum). Valid values from the schema comment are: `STATUS_CHANGE | ASSIGNMENT | RESPONSE | MEDIA_ADDED | SUBSTATUS_CHANGE | PERSON_LINKED | PERSON_UNLINKED | PERSON_ANONYMIZED`. Use these string literals exactly.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client→API | Staff-submitted merge and anonymize requests crossing into /api/staff/people/** |
| db→permanent | Anonymize operation irreversibly nulls PII in persistent DB |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-20 | Elevation of Privilege | POST /api/staff/people/merge — unauthorized merge | mitigate | `requireSession('staff')` as first operation; staff role is minimum required (admin also allowed). Merge is a data operation, not a privilege escalation vector. |
| T-06-21 | Tampering | POST /api/staff/people/merge — IDOR (merging arbitrary person IDs) | mitigate | Route handler verifies both `source` and `target` persons exist via `prisma.person.findFirst({ where: { id, deleted_at: null } })`; returns 404 if either is not found. Staff can see any non-deleted person (CRM is intentionally cross-department for staff). |
| T-06-22 | Denial of Service | POST /api/staff/people/merge — large transaction (many TicketPerson rows) | mitigate | Transaction is bounded by `sourceTicketPersons.length` (total tickets linked to source person); no realistic constituent has thousands of tickets. Transaction is atomic — either all rows migrate or none. |
| T-06-23 | Tampering | PATCH /api/staff/people/[id]/anonymize — re-anonymize attempt | mitigate | Route handler checks `person.anonymized_at !== null` before update; returns `409 ALREADY_ANONYMIZED`. Prevents repeated anonymize calls from adding spurious TicketHistory entries. |
| T-06-24 | Information Disclosure | POST /api/staff/people/merge metadata — PII in TicketHistory notes | mitigate | TicketHistory `note` field in merge and anonymize handlers contains only `"Person records merged by {username}"` and `"Person record anonymized by {username}"` — no PII values included. |
| T-06-25 | Repudiation | Anonymization — no audit trail | mitigate | Anonymize transaction creates TicketHistory entries with `action = 'PERSON_ANONYMIZED'` on every linked ticket, plus `actor_id = session.user.id`. Irreversible + audited. |
</threat_model>
