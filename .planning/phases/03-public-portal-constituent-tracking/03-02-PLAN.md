**1. Create `lib/media.ts`** — Postgres bytea / Large Object media helpers (from TechArch §2.2):

```typescript
// lib/media.ts
// Responsibilities (TechArch §2.2 EXACT):
//   storeMedia(prisma, ticketId, file) — decides bytea vs. LO based on MEDIA_LO_THRESHOLD_KB; creates Media record
//   readMedia(prisma, mediaId) — returns { buffer: Buffer, mimeType, filename } from bytea or LO
//   deleteMedia(prisma, mediaId) — removes LO OID if applicable, deletes Media record
```

Environment variables used:
- `MEDIA_LO_THRESHOLD_KB` (default 8) — files larger than this (in KB) use Postgres Large Object
- `MEDIA_MAX_SIZE_MB` (default 10) — maximum upload size in MB

**storeMedia implementation:**
```typescript
export async function storeMedia(
  prisma: PrismaClient,
  ticketId: string,
  file: File,
): Promise<{ id: string; filename: string; mime_type: string; size_bytes: number; created_at: string }> {
  const maxBytes = (parseInt(process.env.MEDIA_MAX_SIZE_MB ?? '10') * 1024 * 1024);
  if (file.size > maxBytes) {
    throw new Error(`File exceeds maximum size of ${process.env.MEDIA_MAX_SIZE_MB ?? 10}MB`);
  }

  const thresholdBytes = (parseInt(process.env.MEDIA_LO_THRESHOLD_KB ?? '8')) * 1024;
  const buffer = Buffer.from(await file.arrayBuffer());

  if (buffer.length <= thresholdBytes) {
    // Store as bytea
    const media = await prisma.media.create({
      data: {
        ticket_id: ticketId,
        mime_type: file.type || 'application/octet-stream',
        filename: file.name,
        data: buffer,
        lo_oid: null,
        size_bytes: buffer.length,
      },
    });
    return { id: media.id, filename: media.filename, mime_type: media.mime_type, size_bytes: media.size_bytes, created_at: media.created_at.toISOString() };
  } else {
    // Store as Postgres Large Object using raw SQL
    // Use prisma.$executeRaw to create LO and write data
    // lo_creat(-1) returns an OID; then lo_open + lowrite
    const result = await prisma.$queryRaw<[{ lo_oid: number }]>`
      SELECT lo_create(0) AS lo_oid
    `;
    const loOid = result[0].lo_oid;
    // Write data in chunks via lo_put(oid, offset, data)
    await prisma.$executeRaw`SELECT lo_put(${loOid}::oid, 0, ${buffer}::bytea)`;
    const media = await prisma.media.create({
      data: {
        ticket_id: ticketId,
        mime_type: file.type || 'application/octet-stream',
        filename: file.name,
        data: null,
        lo_oid: loOid,
        size_bytes: buffer.length,
      },
    });
    return { id: media.id, filename: media.filename, mime_type: media.mime_type, size_bytes: media.size_bytes, created_at: media.created_at.toISOString() };
  }
}
```

**readMedia implementation:**
```typescript
export async function readMedia(
  prisma: PrismaClient,
  mediaId: string,
): Promise<{ buffer: Buffer; mimeType: string; filename: string }> {
  const media = await prisma.media.findUniqueOrThrow({ where: { id: mediaId } });
  if (media.data) {
    return { buffer: media.data, mimeType: media.mime_type, filename: media.filename };
  }
  if (media.lo_oid) {
    const result = await prisma.$queryRaw<[{ data: Buffer }]>`
      SELECT lo_get(${media.lo_oid}::oid) AS data
    `;
    return { buffer: result[0].data, mimeType: media.mime_type, filename: media.filename };
  }
  throw new Error(`Media ${mediaId} has neither bytea nor Large Object data`);
}
```

**deleteMedia implementation:**
```typescript
export async function deleteMedia(prisma: PrismaClient, mediaId: string): Promise<void> {
  const media = await prisma.media.findUniqueOrThrow({ where: { id: mediaId } });
  if (media.lo_oid) {
    await prisma.$executeRaw`SELECT lo_unlink(${media.lo_oid}::oid)`;
  }
  await prisma.media.delete({ where: { id: mediaId } });
}
```

Implementation:
```typescript
import { NextRequest, NextResponse } from 'next/server';
import prisma from '@/lib/prisma';
import { readMedia } from '@/lib/media';

export async function GET(
  _req: NextRequest,
  { params }: { params: { id: string } },
) {
  try {
    const { buffer, mimeType, filename } = await readMedia(prisma, params.id);
    return new NextResponse(buffer, {
      status: 200,
      headers: {
        'Content-Type': mimeType,
        'Content-Length': String(buffer.length),
        'Content-Disposition': `inline; filename="${encodeURIComponent(filename)}"`,
        'Cache-Control': 'private, max-age=3600',
      },
    });
  } catch (err: unknown) {
    if (err instanceof Error && err.message.includes('No Media found')) {
      return NextResponse.json({ error: 'Media not found' }, { status: 404 });
    }
    return NextResponse.json({ error: 'Failed to retrieve media' }, { status: 500 });
  }
}
```

Note: This endpoint is intentionally public (anyone with the media ID can access it — media IDs are CUID2 and unguessable). Staff-level auth restriction is deferred to Phase 5 if needed.
